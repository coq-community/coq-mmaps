Finite Modular Maps: Red-Black Trees

Author: Pierre Letouzey (Universit√© de Paris - INRIA), adapted from earlier works in Coq Standard Library, see README.md. License: LGPL-2.1-only, see file LICENSE.
This module implements finite maps using Red-Black trees. This code is based on MSetRBT.v in Coq's Stdlib, due initially to Andrew W. Appel, 2011. See initial comment at the beginning of MSetRBT.v.
Note that we only prove here that the operations below preserve the binary search tree invariant (Bst, a.k.a Ok predicate here), but *not* the Red/Black balancing invariant. Indeed, the former is enough to implement the desired interface S, and ensure observational correctness. And proceeding this way is quite lighter. For the proofs of RBT balancing, see MMaps.RBTproofs.

Balancing

We adapt when one side is not a true red-black tree. Both sides have the same black depth.

Definition lbal l k vk r :=
 match l with
 | Rd (Rd a x vx b) y vy c ‚áí Rd (Bk a x vx b) y vy (Bk c k vk r)
 | Rd a x vx (Rd b y vy c) ‚áí Rd (Bk a x vx b) y vy (Bk c k vk r)
 | _ ‚áí Bk l k vk r
 end.

Definition rbal l k vk r :=
 match r with
 | Rd (Rd b y vy c) z vz d ‚áí Rd (Bk l k vk b) y vy (Bk c z vz d)
 | Rd b y vy (Rd c z vz d) ‚áí Rd (Bk l k vk b) y vy (Bk c z vz d)
 | _ ‚áí Bk l k vk r
 end.

A variant of rbal, with reverse pattern order.

Definition rbal' l k vk r :=
 match r with
 | Rd b y vy (Rd c z vz d) ‚áí Rd (Bk l k vk b) y vy (Bk c z vz d)
 | Rd (Rd b y vy c) z vz d ‚áí Rd (Bk l k vk b) y vy (Bk c z vz d)
 | _ ‚áí Bk l k vk r
 end.

Balancing with different black depth. One side is almost a red-black tree, while the other is a true red-black tree, but with black depth + 1. Used in deletion.

Definition lbalS l k vk r :=
 match l with
 | Rd a x vx b ‚áí Rd (Bk a x vx b) k vk r
 | _ ‚áí
   match r with
   | Bk a y vy b ‚áí rbal' l k vk (Rd a y vy b)
   | Rd (Bk a y vy b) z vz c ‚áí Rd (Bk l k vk a) y vy (rbal' b z vz (makeRed c))
   | _ ‚áí Rd l k vk r (* impossible *)
   end
 end.

Definition rbalS l k vk r :=
 match r with
 | Rd b y vy c ‚áí Rd l k vk (Bk b y vy c)
 | _ ‚áí
   match l with
   | Bk a x vx b ‚áí lbal (Rd a x vx b) k vk r
   | Rd a x vx (Bk b y vy c) ‚áí Rd (lbal (makeRed a) x vx b) y vy (Bk c k vk r)
   | _ ‚áí Rd l k vk r (* impossible *)
   end
 end.

Insertion


Fixpoint ins x vx s :=
 match s with
 | Leaf ‚áí Rd Leaf x vx Leaf
 | Node c l y vy r ‚áí
   match K.compare x y with
   | Eq ‚áí Node c l y vx r
   | Lt ‚áí
     match c with
     | Red ‚áí Rd (ins x vx l) y vy r
     | Black ‚áí lbal (ins x vx l) y vy r
     end
   | Gt ‚áí
     match c with
     | Red ‚áí Rd l y vy (ins x vx r)
     | Black ‚áí rbal l y vy (ins x vx r)
     end
   end
 end.

Definition add x vx s := makeBlack (ins x vx s).

Deletion


Fixpoint append (l:t) : t ‚Üí t :=
 match l with
 | Leaf ‚áí fun r ‚áí r
 | Node lc ll lx lv lr ‚áí
   fix append_l (r:t) : t :=
   match r with
   | Leaf ‚áí l
   | Node rc rl rx rv rr ‚áí
     match lc, rc with
     | Red, Red ‚áí
       let lrl := append lr rl in
       match lrl with
       | Rd lr' x v rl' ‚áí Rd (Rd ll lx lv lr') x v (Rd rl' rx rv rr)
       | _ ‚áí Rd ll lx lv (Rd lrl rx rv rr)
       end
     | Black, Black ‚áí
       let lrl := append lr rl in
       match lrl with
       | Rd lr' x v rl' ‚áí Rd (Bk ll lx lv lr') x v (Bk rl' rx rv rr)
       | _ ‚áí lbalS ll lx lv (Bk lrl rx rv rr)
       end
     | Black, Red ‚áí Rd (append_l rl) rx rv rr
     | Red, Black ‚áí Rd ll lx lv (append lr r)
     end
   end
 end.

Fixpoint del x m : t :=
 match m with
 | Leaf ‚áí Leaf
 | Node _ a y v b ‚áí
   match K.compare x y with
   | Eq ‚áí append a b
   | Lt ‚áí
     match a with
     | Bk _ _ _ _ ‚áí lbalS (del x a) y v b
     | _ ‚áí Rd (del x a) y v b
     end
   | Gt ‚áí
     match b with
     | Bk _ _ _ _ ‚áí rbalS a y v (del x b)
     | _ ‚áí Rd a y v (del x b)
     end
   end
 end.

Definition remove x m := makeBlack (del x m).

Tree-ification

We rebuild a tree of size if pred then n-1 else n as soon as the list l has enough elements

Definition klist A := list (key √ó A).
Local Notation treeify_t := (klist elt ‚Üí t √ó klist elt).

Definition bogus : t √ó klist elt := (Leaf, nil).

Definition treeify_zero : treeify_t :=
 fun acc ‚áí (Leaf,acc).

Definition treeify_one : treeify_t :=
 fun acc ‚áí match acc with
 | (x,v)::acc ‚áí (Rd Leaf x v Leaf, acc)
 | _ ‚áí bogus
 end.

Definition treeify_cont (f g : treeify_t) : treeify_t :=
 fun acc ‚áí
 match f acc with
 | (l, (x,v)::acc) ‚áí
   match g acc with
   | (r, acc) ‚áí (Bk l x v r, acc)
   end
 | _ ‚áí bogus
 end.

Fixpoint treeify_aux (pred:bool)(n: positive) : treeify_t :=
 match n with
 | xH ‚áí if pred then treeify_zero else treeify_one
 | xO n ‚áí treeify_cont (treeify_aux pred n) (treeify_aux true n)
 | xI n ‚áí treeify_cont (treeify_aux false n) (treeify_aux pred n)
 end.

Fixpoint plength_aux (l:klist elt)(p:positive) := match l with
 | nil ‚áí p
 | _::l ‚áí plength_aux l (Pos.succ p)
end.

Definition plength (l:klist elt) := plength_aux l 1.

Definition treeify (l:klist elt) :=
 fst (treeify_aux true (plength l) l).

End Elt.

Merge


Section Merge.
Variable elt elt' elt'' : Type.
Variable f : key ‚Üí option elt ‚Üí option elt' ‚Üí option elt''.

Fixpoint mergeL (l1:klist elt) :
 klist elt' ‚Üí klist elt'' ‚Üí klist elt'' :=
 match l1 with
 | nil ‚áí mapoL (fun k v' ‚áí f k None (Some v'))
 | (x,vx)::l1' ‚áí
    fix mergeL_l1 l2 acc :=
    match l2 with
    | nil ‚áí mapoL (fun k v ‚áí f k (Some v) None) l1 acc
    | (y,vy)::l2' ‚áí
       match K.compare x y with
       | Eq ‚áí mergeL l1' l2' (ocons x (f x (Some vx) (Some vy)) acc)
       | Lt ‚áí mergeL_l1 l2' (ocons y (f y None (Some vy)) acc)
       | Gt ‚áí mergeL l1' l2 (ocons x (f x (Some vx) None) acc)
       end
    end
 end.

Definition merge (m1 : t elt) (m2 : t elt') : t elt'' :=
 treeify (mergeL (rev_bindings m1) (rev_bindings m2) nil).

End Merge.

Generic handling for red-matching and red-red-matching


Definition isblack m :=
 match m with Bk _ _ _ _ ‚áí True | _ ‚áí False end.

Definition notblack m :=
 match m with Bk _ _ _ _ ‚áí False | _ ‚áí True end.

Definition notred m :=
 match m with Rd _ _ _ _ ‚áí False | _ ‚áí True end.

Definition rcase {A} f g m : A :=
 match m with
 | Rd a x v b ‚áí f a x v b
 | _ ‚áí g m
 end.

Inductive rspec {A} f g : t elt ‚Üí A ‚Üí Prop :=
 | rred a x v b : rspec f g (Rd a x v b) (f a x v b)
 | relse t : notred t ‚Üí rspec f g t (g t).

Fact rmatch {A} f g m : rspec (A:=A) f g m (rcase f g m).
Proof.
unfold rcase. destmatch; now constructor.
Qed.

Definition rrcase {A} f g m : A :=
 match m with
 | Rd (Rd a x vx b) y vy c ‚áí f a x vx b y vy c
 | Rd a x vx (Rd b y vy c) ‚áí f a x vx b y vy c
 | _ ‚áí g m
 end.

Notation notredred := (rrcase (fun _ _ _ _ _ _ _ ‚áí False) (fun _ ‚áí True)).

Inductive rrspec {A} f g : t elt ‚Üí A ‚Üí Prop :=
 | rrleft a x vx b y vy c :
     rrspec f g (Rd (Rd a x vx b) y vy c) (f a x vx b y vy c)
 | rrright a x vx b y vy c :
     rrspec f g (Rd a x vx (Rd b y vy c)) (f a x vx b y vy c)
 | rrelse m : notredred m ‚Üí rrspec f g m (g m).

Fact rrmatch {A} f g m : rrspec (A:=A) f g m (rrcase f g m).
Proof.
unfold rrcase. destmatch; now constructor.
Qed.

Definition rrcase' {A} f g m : A :=
 match m with
 | Rd a x vx (Rd b y vy c) ‚áí f a x vx b y vy c
 | Rd (Rd a x vx b) y vy c ‚áí f a x vx b y vy c
 | _ ‚áí g m
 end.

Fact rrmatch' {A} f g m : rrspec (A:=A) f g m (rrcase' f g m).
Proof.
unfold rrcase'. destmatch; now constructor.
Qed.

Fact lbal_match l k v r :

   (fun a x vx b y vy c ‚áí Rd (Bk a x vx b) y vy (Bk c k v r))
   (fun l ‚áí Bk l k v r)
   l
   (lbal l k v r).

Qed.

 rrspec
   (fun a x vx b y vy c ‚áí Rd (Bk l k v a) x vx (Bk b y vy c))
   (fun r ‚áí Bk l k v r)
   r
   (rbal l k v r).
Proof.
 exact (rrmatch _ _ _).
Qed.

 rrspec
   (fun a x vx b y vy c ‚áí Rd (Bk l k v a) x vx (Bk b y vy c))
   (fun r ‚áí Bk l k v r)

  (fun a y vy b ‚áí Rd (Bk a y vy b) x vx r)

Definition treeify_invariant size (f:treeify_t) :=
 ‚àÄ acc,
 size ‚â§ length acc ‚Üí
 let (m,acc') := f acc in
 cardinal m = size ‚àß acc = bindings m ++ acc'.

Lemma treeify_zero_spec : treeify_invariant 0 (@treeify_zero elt).
Proof.
 intro. simpl. auto.
Qed.

Lemma treeify_one_spec : treeify_invariant 1 (@treeify_one elt).
Proof.
 intros [|x acc]; simpl; auto. inversion 1.
 destruct x. simpl; auto.
Qed.

Lemma treeify_cont_spec f g size1 size2 size :
 treeify_invariant size1 f ‚Üí
 treeify_invariant size2 g ‚Üí
 size = S (size1 + size2) ‚Üí
 treeify_invariant size (treeify_cont f g).
Proof.
 intros Hf Hg EQ acc LE. unfold treeify_cont.
 specialize (Hf acc).
 destruct (f acc) as (t1,acc1).
 destruct Hf as (Hf1,Hf2).
  { transitivity size; trivial. subst. auto with arith. }
 destruct acc1 as [|x acc1].
  { exfalso. revert LE. apply Nat.lt_nge. subst.
    rewrite app_nil_r, <- cardinal_spec; auto with arith. }
 specialize (Hg acc1).
 destruct (g acc1) as (t2,acc2).
 destruct Hg as (Hg1,Hg2).
  { revert LE. subst.
    rewrite app_length, <- cardinal_spec. simpl.
    rewrite Nat.add_succ_r, <- Nat.succ_le_mono.
    apply Nat.add_le_mono_l. }
 destruct x; simpl.
 rewrite bindings_node_acc. subst; auto.
Qed.

Lemma treeify_aux_spec n (p:bool) :
 treeify_invariant (ifpred p (Pos.to_nat n)) (treeify_aux p n).
Proof.
 revert p.
 induction n as [n|n|]; intros p; simpl treeify_aux.
 - eapply treeify_cont_spec; [ apply (IHn false) | apply (IHn p) | ].
   rewrite Pos2Nat.inj_xI.
   assert (H := Pos2Nat.is_pos n). apply Nat.neq_0_lt_0 in H.
   destruct p; simpl; intros; rewrite Nat.add_0_r; trivial.
   now rewrite <- Nat.add_succ_r, Nat.succ_pred; trivial.
 - eapply treeify_cont_spec; [ apply (IHn p) | apply (IHn true) | ].
   rewrite Pos2Nat.inj_xO.
   assert (H := Pos2Nat.is_pos n). apply Nat.neq_0_lt_0 in H.
   rewrite <- Nat.add_succ_r, Nat.succ_pred by trivial.
   destruct p; simpl; intros; rewrite Nat.add_0_r; trivial.
   symmetry. now apply Nat.add_pred_l.
 - destruct p; [ apply treeify_zero_spec | apply treeify_one_spec ].
Qed.
